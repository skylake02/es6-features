"WTF is ECMAScript? Where did JavaScript go?"
Difference Between Javascript and ECMAscript

tl;dr

Javascript is a "dialect" of ECMAScript.
ECMAScript is a general purpose scripting language.
JavaScript, on the other hand, is based on ECMAScript, but is a scripting language commonly run in browsers.

ECMAScript 6 has some new and shiny features to help write more concise scripts.

const and let vs. var

const foo = true;
// variable is unchangeable upon binding

let bar = false;
// similar to var but not completely

let respects block scope, while var does not

let foo = 5;

if (true) {
  let foo = 10;
  console.log(foo);
  // 10
}

console.log(foo);
// 5 var foo = 5;

if (true) {
  var foo = 10;
  console.log(foo);
  // 10
}

console.log(foo);
// 10

default function parameters

parameters can now have a default value if one is not passed to a function that asks for one

old way

function add (a, b) {
  if (a === undefined) {
    a = 2;
  }
  if (b === undefined) {
    b = 1;
  }
  return a + b;
}

console.log(add());
// 3

new way

function add (a=2, b=1) {
  return a + b;
}

console.log(add());
// 3

the spread operator (...)

an operator that expands iteratable data structures (e.g. arrays)

function sum (a, b, c) {
  return a + b + c;
}

let arr1 = [1, 2, 3];

console.log(...arr1);
// 1 2 3
// these are not part of a string either
// they are numbers

console.log(sum(...arr1));
// 6
// the array can be plugged in as arguments

console.log([...arr1, 4, 5, 6])
// [1, 2, 3, 4, 5, 6]
// a better way to concatenate

object literal enhancements

old way

var name = 'Jon',
    age = 21,
    man = {
        name: name,
        age: age,
        say: function (msg) {
            console.log(msg);
        };
     };

console.log(man.name);
// 'Jon'
console.log(man.say('Hello, sir!'));
// 'Hello, sir!'

new way

let name = 'Jon',
    age = 21,
    man = {
        name, age,
        say (msg) {
            console.log(msg;
        }
     };

console.log(man.name);
// 'Jon'
console.log(man.say('Hello, sir!'));
// 'Hello, sir!'

template strings (`back ticks`)

a better way to embed variables into strings

let name = 'Sam';

console.log('Hello, I am ' + name + '! Nice to meet you!');
// 'Hello, I am Sam! Nice to meet you!'

console.log(`Greetings! My name is ${name}! How are you?`);
// 'Greetings! My name is Sam! How are you?'

arrow functions (=>)

a different way to write functions

no parameters

let greet = function() {
  console.log('Hello!');
}

greet();
// 'Hello!' let greet = () => {
  console.log('Greetings!');
}

greet();
// 'Greetings!' let greet = () => console.log('Greetings!');

greet();
// 'Greetings!'

with parameters

let greet = function(name) {
  console.log(`Hello, ${name}.`);
}

greet('Dan');
// 'Hello, Dan.' let greet = (name) => console.log(`Greetings, ${name}.`);

greet('Bob');
// 'Greetings, Bob.'

returning

let greet = function() {
  return 'Hello!';
}

console.log(greet());
// 'Hello!' let greet = () => {
  return 'Hello!'
};

console.log(greet());
// 'Hello!' let greet = () => 'Hello!';

console.log(greet());
// 'Hello!'

return isn't even needed

lexical this

this refers to the this of the encapsulating function

let boi = {
  name: 'Frank',
  run(x) {
    let _this = this;
    window.setInterval(function() {
      console.log(`I, ${_this.name}, ran ${x} meters!`);
    }, 1000);
  }
}

boi.run(100);
// 'I, Frank, ran 100 meters!'

a _this variable has to be made to reach outside the scope of the setInterval callback function

let boi = {
  name: 'Frank',
  run(x) {
    window.setInterval(() => {
      console.log(`I, ${this.name}, ran ${x} meters!`);
    }, 1000);
  }
}

boi.run(100);
// 'I, Frank, ran 100 meters!'

however, arrow functions do not bind their own this, and therefore have this set to their encapsulating function

new string methods

String.repeat()

let str = 'Whoa dude! '

console.log(str.repeat(5));
// 'Whoa dude! Whoa dude! Whoa dude! Whoa dude! Whoa dude!'

String.startsWith('')

let str = 'Goodbye, mate.'

console.log(str.startsWith('Good'));
// true

the starting position can also be passed in, with the beginning starting at 0

let str = 'Goodbye, mate.'

console.log(str.startsWith('mate', 9));
// true

String.endsWith('')

same thing here, just the reverse

let str = 'Goodbye, mate.';

console.log(str.endsWith('mate.', str.length));
// true let str = 'Goodbye, mate.';

console.log(str.endsWith('mate.'));
// true

the two are the same

let str = 'Goodbye, mate.'

console.log(str.endsWith('Goodbye', str.length - 7));
// true

String.includes('')

let str = 'My name is Jack!';

console.log(str.includes('Jack'));
// true

sets

a new datatype

Set.add(), Set.size(), Set.delete()

let names = new Set();

names.add('Tom').add('Jak').add('Crystal');
console.log(names);
// {'Tom', 'Jak', 'Crystal'}


names.add('Crystal');
console.log(names);
// {'Tom', 'Jak', 'Crystal'}
// duplicates cannot be made


console.log(names.size);
// 3


names.delete('Tom');
console.log(names);
// {'Jak', 'Crystal'}

don't chain .delete()s though

let names = new Set();

names.add('Tom').add('Jak').add('Crystal');

console.log(names.delete('Tom'));
// true

names.add('Tom');
names.delete('Tom').delete('Jak');
// *Uncaught TypeError* let names = new Set();

names.add('Tom').add('Jak').add('Crystal');

names.clear();
console.log(names.size);
// 0
console.log(names);
// {}

Set to Array

let dudes = ['Josh', 'Ava', 'Sally', 'Talia', 'Haley'];

// converting to a set
let realDudes = new Set(dudes);

console.log(realDudes);
// {'Josh', 'Ava', 'Sally', 'Talia', 'Haley'}

// converting to an array
console.log([...realDudes]);
// ['Josh', 'Ava', 'Sally', 'Talia', 'Haley']
